% I set colors to 50% throughout so they will work on a black and white printer
\colorlet{darkgreen}{green!50!black}
% Create a running 12 month calendar starting from the current month
\calendar[dates=\year-\month-01 to \year-\month-01+365,week list,
month label left vertical,month yshift=0pt,
month text=\textcolor{darkgreen}{\%mt}]
% Weekends are grey
if (weekend) [black!50]
% Regular holidays are red
if (equals=01-01,
    equals=07-04,
    equals=12-24,
    equals=12-25,
    equals=12-31]) [red!50]
% Holidays that vary by year are red
if (equals=2020-01-20,
    equals=2020-04-10,
    equals=2020-05-25,
    equals=2020-09-07,
    equals=2020-11-26,
    equals=2021-01-18,
    equals=2021-04-02,
    equals=2021-05-31,
    equals=2021-09-06,
    equals=2021-11-25) [red!50];
