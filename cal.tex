month text=\textcolor{darkgreen}{\%mt}]
% Weekends are grey

% Regular holidays are red
if (equals=01-01,
    equals=07-04,
    equals=12-24,
    equals=12-25,
    equals=12-31]) [red!50]
% Holidays that vary by year are red
if (equals=2022-01-17,
    equals=2022-04-15,
    equals=2022-05-30,
    equals=2022-09-05,
    equals=2022-11-24,
    equals=2023-01-16,
