\calendar[dates=\year-\month-01 to \year-\month-01+365,week list,

month text=\textcolor{darkgreen}{\%mt}]
% Weekends are grey

% Regular holidays are red
if (
    equals=01-01,
    equals=07-04,
    equals=12-24,
    equals=12-25,
    equals=12-31,
   ) [red!50]
% Holidays that vary by year are red
if (
    equals=2024-02-19,
    equals=2024-03-29,
    equals=2024-05-27,
    equals=2024-06-19,
    equals=2024-09-02,
    equals=2024-11-28,
